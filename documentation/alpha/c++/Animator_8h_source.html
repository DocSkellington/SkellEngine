218  inline void update(sf::Uint64 deltaTime) {
219  if (!m_pause) {
220  // How much time we have left to execute
221  sf::Time durationLeft = sf::Time::Zero;
222 
223  // First, we clear the list of playing animations
224  if (!m_playingAnimations.empty()) {
225  // Since it's a queue, we know that the last animation in the queue is the last played animation
226  sf::Time lastAnimationDuration = m_playingAnimations.back().getDuration();
227  if (m_elapsedTime < lastAnimationDuration) {
228  // We stille have to update the last animation. Therefore, we remove everything except the last one
229  m_playingAnimations.erase(m_playingAnimations.begin(), std::prev(m_playingAnimations.end()));
230  durationLeft = lastAnimationDuration;
231  }
232  else {
233  // The last animation was fully executed. We can remove everything
234  m_playingAnimations.clear();
235  }
236  }
237 
238  // We add the deltaTime to the elapsed time of the animator
239  m_elapsedTime += sf::microseconds(deltaTime);
240 
241  // The last played animation lefts enough time to start a new animation
242  while (m_elapsedTime >= durationLeft) {
243  // If the last animation must loop, we add it at the first position in the queued animations
244  if (!m_playingAnimations.empty() && m_playingAnimations.back().isLoop()) {
245  m_queuedAnimations.push_front(m_playingAnimations.back());
246  }
247 
248  if (m_queuedAnimations.empty()) {
249  // We can't directly set to Time::Zero because of the animate function that uses m_elapsedTime
250  // So, we set the variable to the left duration
251  // If there is no animation at all for a frame, m_elapsedTime goes back to Time::Zero
252  m_elapsedTime = durationLeft;
253  break;
254  }
255  else {
256  // We retrieve the next animation and start to play it
257  auto nextAnimation = m_queuedAnimations.front();
258  m_queuedAnimations.pop_front();
259  m_playingAnimations.push_back(nextAnimation);
260 
261  // We update duration and the elapsed time to match the new animation
262  m_elapsedTime -= durationLeft; // We remove the time it took until here
263  durationLeft = nextAnimation.getDuration();
264  }
265  }
266  }
267  }

269  inline void animate(Animated &animated) {
270  if (!m_pause && !m_playingAnimations.empty()) {
271  // For every playing animation except the last one, we know that the animation is fully executed
272  // So, we just call the animation function with a progress of 1
273  for (auto itr = m_playingAnimations.begin() ; itr != std::prev(m_playingAnimations.end()) ; ++itr) {
274  (*itr)(animated, 1.);
275  }
276 
277  // For the last animation, we need to compute the exact progress
278  auto& animation = m_playingAnimations.back();
279  animation(animated, m_elapsedTime / animation.getDuration());
280  }
