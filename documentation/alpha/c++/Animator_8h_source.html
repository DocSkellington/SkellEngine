22  void update(sf::Uint64 deltaTime) {
23  if (!m_pause) {
24  // How much time we have left to execute
25  sf::Time durationLeft = sf::Time::Zero;
26 
27  // First, we clear the list of playing animations
28  if (!m_playingAnimations.empty()) {
29  // Since it's a queue, we know that the last animation in the queue is the last played animation
30  sf::Time lastAnimationDuration = m_playingAnimations.back().getDuration();
31  if (m_elapsedTime < lastAnimationDuration) {
32  // We stille have to update the last animation. Therefore, we remove everything except the last one
33  m_playingAnimations.erase(m_playingAnimations.begin(), std::prev(m_playingAnimations.end()));
34  durationLeft = lastAnimationDuration;
35  }
36  else {
37  // The last animation was fully executed. We can remove everything
38  m_playingAnimations.clear();
39  }
40  }
41 
42  // We add the deltaTime to the elapsed time of the animator
43  m_elapsedTime += sf::microseconds(deltaTime);
44 
45  // The last played animation lefts enough time to start a new animation
46  while (m_elapsedTime >= durationLeft) {
47  // If the last animation must loop, we add it at the first position in the queued animations
48  if (!m_playingAnimations.empty() && m_playingAnimations.back().isLoop()) {
49  m_queuedAnimations.push_front(m_playingAnimations.back());
50  }
51 
52  if (m_queuedAnimations.empty()) {
53  // We can't directly set to Time::Zero because of the animate function that uses m_elapsedTime
54  // So, we set the variable to the left duration
55  // If there is no animation at all for a frame, m_elapsedTime goes back to Time::Zero
56  m_elapsedTime = durationLeft;
57  break;
58  }
59  else {
60  // We retrieve the next animation and start to play it
61  auto nextAnimation = m_queuedAnimations.front();
62  m_queuedAnimations.pop_front();
63  m_playingAnimations.push_back(nextAnimation);
64 
65  // We update duration and the elapsed time to match the new animation
66  m_elapsedTime -= durationLeft; // We remove the time it took until here
67  durationLeft = nextAnimation.getDuration();
68  }
